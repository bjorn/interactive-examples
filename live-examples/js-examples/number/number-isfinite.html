function div(x, y) {
  if (Number.isFinite(x / y)) {
    return 'Result is Finite.';
  }
  return 'Result is NOT Finite!';
}

console.log(div(1, 0));
// expected output: "Result is NOT Finite!"

console.log(div(10, 5));
// expected output: "Result is Finite."

console.log(div(0, 0));
// expected output: "Result is NOT Finite!"
